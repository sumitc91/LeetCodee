468. Validate IP Address

Jump to Solution: Python Java C++ JavaScript C#

Problem Description

Given a string queryIP, return "IPv4" if IP is a valid IPv4 address, "IPv6" if IP is a valid IPv6 address or "Neither" if IP is not a correct IP of any type.

A valid IPv4 address is an IP in the form "x1.x2.x3.x4" where 0 ≤ xi ≤ 255 and xi cannot contain leading zeros. For example: "192.168.1.1" and "192.168.1.0" are valid IPv4 addresses while "192.168.01.1", "192.168.1.00", and "192.168@1.1" are invalid IPv4 addresses.

A valid IPv6 address is an IP in the form "x1:x2:x3:x4:x5:x6:x7:x8" where:

  • 1 ≤ xi.length ≤ 4
  • xi is a hexadecimal string which may contain digits, lowercase English letters ('a' to 'f'), and uppercase English letters ('A' to 'F')
  • Leading zeros are allowed in xi

For example: "2001:0db8:85a3:0000:0000:8a2e:0370:7334" and "2001:db8:85a3:0:0:8A2E:0370:7334" are valid IPv6 addresses, while "2001:0db8:85a3::8A2E:037j:7334" and "02001:0db8:85a3:0000:0000:8a2e:0370:7334" are invalid IPv6 addresses.

Example 1:

Input: queryIP = "172.16.254.1"
Output: "IPv4"
Explanation: This is a valid IPv4 address, return "IPv4".

Example 2:

Input: queryIP = "2001:0db8:85a3:0:0:8A2E:0370:7334"
Output: "IPv6"
Explanation: This is a valid IPv6 address, return "IPv6".

Example 3:

Input: queryIP = "256.256.256.256"
Output: "Neither"
Explanation: This is neither a valid IPv4 address nor a valid IPv6 address.

Constraints:

  • queryIP consists only of English letters, digits and the characters '.' and ':'.

Solution

Python Solution

class Solution:
    def validIPAddress(self, queryIP: str) -> str:
        def isIPv4(s):
            try:
                # Split by dots and check if we have exactly 4 parts
                parts = s.split('.')
                if len(parts) != 4:
                    return False
                
                # Check each part
                for part in parts:
                    # Check for empty parts or parts with length > 3
                    if len(part) == 0 or len(part) > 3:
                        return False
                    # Check for leading zeros
                    if len(part) > 1 and part[0] == '0':
                        return False
                    # Check if all characters are digits
                    if not part.isdigit():
                        return False
                    # Check if value is between 0 and 255
                    if not 0 <= int(part) <= 255:
                        return False
                return True
            except:
                return False
        
        def isIPv6(s):
            try:
                # Split by colons and check if we have exactly 8 parts
                parts = s.split(':')
                if len(parts) != 8:
                    return False
                
                # Check each part
                for part in parts:
                    # Check for empty parts or parts with length > 4
                    if len(part) == 0 or len(part) > 4:
                        return False
                    # Check if all characters are valid hex digits
                    for c in part.lower():
                        if not (c.isdigit() or 'a' <= c <= 'f'):
                            return False
                return True
            except:
                return False
        
        if isIPv4(queryIP):
            return "IPv4"
        if isIPv6(queryIP):
            return "IPv6"
        return "Neither"

Java Solution

class Solution {
    public String validIPAddress(String queryIP) {
        if (isIPv4(queryIP)) return "IPv4";
        if (isIPv6(queryIP)) return "IPv6";
        return "Neither";
    }
    
    private boolean isIPv4(String ip) {
        try {
            // Split by dots and check if we have exactly 4 parts
            String[] parts = ip.split("\\.", -1);
            if (parts.length != 4) return false;
            
            // Check each part
            for (String part : parts) {
                // Check for empty parts or parts with length > 3
                if (part.length() == 0 || part.length() > 3) return false;
                // Check for leading zeros
                if (part.length() > 1 && part.charAt(0) == '0') return false;
                // Check if all characters are digits
                for (char c : part.toCharArray()) {
                    if (!Character.isDigit(c)) return false;
                }
                // Check if value is between 0 and 255
                int value = Integer.parseInt(part);
                if (value < 0 || value > 255) return false;
            }
            return true;
        } catch (Exception e) {
            return false;
        }
    }
    
    private boolean isIPv6(String ip) {
        try {
            // Split by colons and check if we have exactly 8 parts
            String[] parts = ip.split(":", -1);
            if (parts.length != 8) return false;
            
            // Check each part
            for (String part : parts) {
                // Check for empty parts or parts with length > 4
                if (part.length() == 0 || part.length() > 4) return false;
                // Check if all characters are valid hex digits
                for (char c : part.toLowerCase().toCharArray()) {
                    if (!Character.isDigit(c) && (c < 'a' || c > 'f')) return false;
                }
            }
            return true;
        } catch (Exception e) {
            return false;
        }
    }
}

class Solution {
public:
    string validIPAddress(string queryIP) {
        if (isIPv4(queryIP)) return "IPv4";
        if (isIPv6(queryIP)) return "IPv6";
        return "Neither";
    }
    
private:
    bool isIPv4(string& ip) {
        // Split by dots and check if we have exactly 4 parts
        vector parts;
        stringstream ss(ip);
        string part;
        while (getline(ss, part, '.')) {
            parts.push_back(part);
        }
        if (parts.size() != 4 || ip.back() == '.') return false;
        
        // Check each part
        for (const string& part : parts) {
            // Check for empty parts or parts with length > 3
            if (part.empty() || part.length() > 3) return false;
            // Check for leading zeros
            if (part.length() > 1 && part[0] == '0') return false;
            // Check if all characters are digits
            for (char c : part) {
                if (!isdigit(c)) return false;
            }
            // Check if value is between 0 and 255
            int value = stoi(part);
            if (value < 0 || value > 255) return false;
        }
        return true;
    }
    
    bool isIPv6(string& ip) {
        // Split by colons and check if we have exactly 8 parts
        vector parts;
        stringstream ss(ip);
        string part;
        while (getline(ss, part, ':')) {
            parts.push_back(part);
        }
        if (parts.size() != 8 || ip.back() == ':') return false;
        
        // Check each part
        for (const string& part : parts) {
            // Check for empty parts or parts with length > 4
            if (part.empty() || part.length() > 4) return false;
            // Check if all characters are valid hex digits
            for (char c : part) {
                c = tolower(c);
                if (!isdigit(c) && (c < 'a' || c > 'f')) return false;
            }
        }
        return true;
    }
};

/**
 * @param {string} queryIP
 * @return {string}
 */
var validIPAddress = function(queryIP) {
    const isIPv4 = (ip) => {
        try {
            // Split by dots and check if we have exactly 4 parts
            const parts = ip.split('.');
            if (parts.length !== 4) return false;
            
            // Check each part
            for (const part of parts) {
                // Check for empty parts or parts with length > 3
                if (part.length === 0 || part.length > 3) return false;
                // Check for leading zeros
                if (part.length > 1 && part[0] === '0') return false;
                // Check if all characters are digits
                if (!/^\d+$/.test(part)) return false;
                // Check if value is between 0 and 255
                const value = parseInt(part);
                if (value < 0 || value > 255) return false;
            }
            return true;
        } catch {
            return false;
        }
    };
    
    const isIPv6 = (ip) => {
        try {
            // Split by colons and check if we have exactly 8 parts
            const parts = ip.split(':');
            if (parts.length !== 8) return false;
            
            // Check each part
            for (const part of parts) {
                // Check for empty parts or parts with length > 4
                if (part.length === 0 || part.length > 4) return false;
                // Check if all characters are valid hex digits
                if (!/^[0-9a-fA-F]+$/.test(part)) return false;
            }
            return true;
        } catch {
            return false;
        }
    };
    
    if (isIPv4(queryIP)) return "IPv4";
    if (isIPv6(queryIP)) return "IPv6";
    return "Neither";
};

C# Solution

public class Solution {
    public string ValidIPAddress(string queryIP) {
        if (IsIPv4(queryIP)) return "IPv4";
        if (IsIPv6(queryIP)) return "IPv6";
        return "Neither";
    }
    
    private bool IsIPv4(string ip) {
        try {
            // Split by dots and check if we have exactly 4 parts
            string[] parts = ip.Split('.');
            if (parts.Length != 4) return false;
            
            // Check each part
            foreach (string part in parts) {
                // Check for empty parts or parts with length > 3
                if (part.Length == 0 || part.Length > 3) return false;
                // Check for leading zeros
                if (part.Length > 1 && part[0] == '0') return false;
                // Check if all characters are digits
                if (!part.All(char.IsDigit)) return false;
                // Check if value is between 0 and 255
                int value = int.Parse(part);
                if (value < 0 || value > 255) return false;
            }
            return true;
        } catch {
            return false;
        }
    }
    
    private bool IsIPv6(string ip) {
        try {
            // Split by colons and check if we have exactly 8 parts
            string[] parts = ip.Split(':');
            if (parts.Length != 8) return false;
            
            // Check each part
            foreach (string part in parts) {
                // Check for empty parts or parts with length > 4
                if (part.Length == 0 || part.Length > 4) return false;
                // Check if all characters are valid hex digits
                if (!part.All(c => char.IsDigit(c) || 
                    ('a' <= char.ToLower(c) && char.ToLower(c) <= 'f'))) {
                    return false;
                }
            }
            return true;
        } catch {
            return false;
        }
    }
}
