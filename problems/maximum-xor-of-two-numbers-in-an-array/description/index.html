421. Maximum XOR of Two Numbers in an Array

Jump to Solution: Python Java C++ JavaScript C#

Problem Description

Given an integer array nums, return the maximum result of nums[i] XOR nums[j], where 0 ≤ i ≤ j < n.

Example 1:

Input: nums = [3,10,5,25,2,8]
Output: 28
Explanation: The maximum result is 5 XOR 25 = 28.

Example 2:

Input: nums = [14,70,53,83,49,91,36,80,92,51,66,70]
Output: 127

Constraints:

  • 1 <= nums.length <= 2 * 10⁵
  • 0 <= nums[i] <= 2³¹ - 1

Solution

Python Solution

class Solution:
    def findMaximumXOR(self, nums: List[int]) -> int:
        # Length of max number in binary
        L = len(bin(max(nums))) - 2
        max_xor = 0
        
        # Try to find the maximum XOR for each bit position
        for i in range(L-1, -1, -1):
            # Go to the next bit by shifting
            max_xor <<= 1
            curr_xor = max_xor | 1
            
            # Compute all possible prefixes
            prefixes = {num >> i for num in nums}
            
            # Check if we can get curr_xor
            for p in prefixes:
                if p ^ (curr_xor >> 1) in prefixes:
                    max_xor = curr_xor
                    break
                    
        return max_xor

Java Solution

class Solution {
    public int findMaximumXOR(int[] nums) {
        int maxNum = 0;
        for (int num : nums) {
            maxNum = Math.max(maxNum, num);
        }
        
        // Length of max number in binary
        int L = Integer.toBinaryString(maxNum).length();
        int maxXor = 0;
        
        // Try to find the maximum XOR for each bit position
        for (int i = L - 1; i >= 0; i--) {
            // Go to the next bit by shifting
            maxXor <<= 1;
            int currXor = maxXor | 1;
            
            // Compute all possible prefixes
            Set prefixes = new HashSet<>();
            for (int num : nums) {
                prefixes.add(num >> i);
            }
            
            // Check if we can get currXor
            for (int p : prefixes) {
                if (prefixes.contains(p ^ (currXor >> 1))) {
                    maxXor = currXor;
                    break;
                }
            }
        }
        
        return maxXor;
    }
}

public class Solution {
    public int FindMaximumXOR(int[] nums) {
        int maxNum = nums.Max();
        
        // Length of max number in binary
        int L = Convert.ToString(maxNum, 2).Length;
        int maxXor = 0;
        
        // Try to find the maximum XOR for each bit position
        for (int i = L - 1; i >= 0; i--) {
            // Go to the next bit by shifting
            maxXor <<= 1;
            int currXor = maxXor | 1;
            
            // Compute all possible prefixes
            HashSet prefixes = new HashSet();
            foreach (int num in nums) {
                prefixes.Add(num >> i);
            }
            
            // Check if we can get currXor
            foreach (int p in prefixes) {
                if (prefixes.Contains(p ^ (currXor >> 1))) {
                    maxXor = currXor;
                    break;
                }
            }
        }
        
        return maxXor;
    }
}
