391. Perfect Rectangle

Jump to Solution: Python Java C++ JavaScript C#

Problem Description

Given an array rectangles where rectangles[i] = [xi, yi, ai, bi] represents an axis-aligned rectangle. The bottom-left point of the rectangle is (xi, yi) and the top-right point of it is (ai, bi).

Return true if all the rectangles together form an exact cover of a rectangular region.

Example 1:

Input: rectangles = [[1,1,3,3],[3,1,4,2],[3,2,4,4],[1,3,2,4],[2,3,3,4]]
Output: true
Explanation: All 5 rectangles together form an exact cover of a rectangular region.

Example 2:

Input: rectangles = [[1,1,2,3],[1,3,2,4],[3,1,4,2],[3,2,4,4]]
Output: false
Explanation: Because there is a gap between the two rectangular regions.

Example 3:

Input: rectangles = [[1,1,3,3],[3,1,4,2],[1,3,2,4],[2,2,4,4]]
Output: false
Explanation: Because two of the rectangles overlap with each other.

Constraints:

  • 1 <= rectangles.length <= 2 * 10⁴
  • rectangles[i].length == 4
  • -10⁵ <= xi, yi, ai, bi <= 10⁵

Solution

Python Solution

class Solution:
    def isRectangleCover(self, rectangles: List[List[int]]) -> bool:
        # Set to store corners
        corners = set()
        
        # Calculate area and find corners
        area = 0
        min_x = min_y = float('inf')
        max_x = max_y = float('-inf')
        
        for x, y, a, b in rectangles:
            # Update boundaries
            min_x = min(min_x, x)
            min_y = min(min_y, y)
            max_x = max(max_x, a)
            max_y = max(max_y, b)
            
            # Calculate area
            area += (a - x) * (b - y)
            
            # Add/remove corners
            for corner in [(x, y), (x, b), (a, y), (a, b)]:
                if corner in corners:
                    corners.remove(corner)
                else:
                    corners.add(corner)
        
        # Check if remaining corners are the four corners of the perfect rectangle
        if len(corners) != 4:
            return False
            
        # Check if corners match the boundaries
        if not {(min_x, min_y), (min_x, max_y), (max_x, min_y), (max_x, max_y)} == corners:
            return False
            
        # Check if area matches
        return area == (max_x - min_x) * (max_y - min_y)

class Solution {
    public boolean isRectangleCover(int[][] rectangles) {
        // Set to store corners
        Set corners = new HashSet<>();
        
        // Calculate area and find corners
        long area = 0;
        int min_x = Integer.MAX_VALUE;
        int min_y = Integer.MAX_VALUE;
        int max_x = Integer.MIN_VALUE;
        int max_y = Integer.MIN_VALUE;
        
        for (int[] rect : rectangles) {
            // Update boundaries
            min_x = Math.min(min_x, rect[0]);
            min_y = Math.min(min_y, rect[1]);
            max_x = Math.max(max_x, rect[2]);
            max_y = Math.max(max_y, rect[3]);
            
            // Calculate area
            area += (long)(rect[2] - rect[0]) * (rect[3] - rect[1]);
            
            // Add/remove corners
            String[] corners_arr = {
                rect[0] + "," + rect[1],
                rect[0] + "," + rect[3],
                rect[2] + "," + rect[1],
                rect[2] + "," + rect[3]
            };
            
            for (String corner : corners_arr) {
                if (!corners.add(corner)) {
                    corners.remove(corner);
                }
            }
        }
        
        // Check if remaining corners are the four corners of the perfect rectangle
        if (corners.size() != 4) {
            return false;
        }
        
        // Check if corners match the boundaries
        Set expected = new HashSet<>(Arrays.asList(
            min_x + "," + min_y,
            min_x + "," + max_y,
            max_x + "," + min_y,
            max_x + "," + max_y
        ));
        
        if (!corners.equals(expected)) {
            return false;
        }
        
        // Check if area matches
        return area == (long)(max_x - min_x) * (max_y - min_y);
    }
}

        min_x = Math.min(min_x, x);
        min_y = Math.min(min_y, y);

            `${a},${b}`

            if (corners.has(corner)) {
                corners.delete(corner);
            } else {
                corners.add(corner);
            }

public class Solution {
    public bool IsRectangleCover(int[][] rectangles) {
        // Set to store corners
        HashSet corners = new HashSet();
        
        // Calculate area and find corners
        long area = 0;
        int min_x = int.MaxValue;
        int min_y = int.MaxValue;
        int max_x = int.MinValue;
        int max_y = int.MinValue;
        
        foreach (var rect in rectangles) {
            // Update boundaries
            min_x = Math.Min(min_x, rect[0]);
            min_y = Math.Min(min_y, rect[1]);
            max_x = Math.Max(max_x, rect[2]);
            max_y = Math.Max(max_y, rect[3]);
            
            // Calculate area
            area += (long)(rect[2] - rect[0]) * (rect[3] - rect[1]);
            
            // Add/remove corners
            string[] corners_arr = {
                $"{rect[0]},{rect[1]}",
                $"{rect[0]},{rect[3]}",
                $"{rect[2]},{rect[1]}",
                $"{rect[2]},{rect[3]}"
            };
            
            foreach (var corner in corners_arr) {
                if (!corners.Add(corner)) {
                    corners.Remove(corner);
                }
            }
        }
        
        // Check if remaining corners are the four corners of the perfect rectangle
        if (corners.Count != 4) {
            return false;
        }
        
        // Check if corners match the boundaries
        HashSet expected = new HashSet {
            $"{min_x},{min_y}",
            $"{min_x},{max_y}",
            $"{max_x},{min_y}",
            $"{max_x},{max_y}"
        };
        
        if (!corners.SetEquals(expected)) {
            return false;
        }
        
        // Check if area matches
        return area == (long)(max_x - min_x) * (max_y - min_y);
    }
}
