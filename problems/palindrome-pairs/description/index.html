336. Palindrome Pairs

Jump to Solution: Python Java C++ JavaScript C#

Problem Description

You are given a list of unique words, return all the pairs of the distinct indices (i, j) in the given list, so that the concatenation of the two words words[i] + words[j] is a palindrome.

Example 1:

Input: words = ["abcd","dcba","lls","s","sssll"]
Output: [[0,1],[1,0],[3,2],[2,4]]
Explanation: The palindromes are ["abcddcba","dcbaabcd","slls","llssssll"]

Example 2:

Input: words = ["bat","tab","cat"]
Output: [[0,1],[1,0]]
Explanation: The palindromes are ["battab","tabbat"]

Example 3:

Input: words = ["a",""]
Output: [[0,1],[1,0]]
Explanation: The palindromes are ["a","a"]

Constraints:

  • 1 <= words.length <= 5000
  • 0 <= words[i].length <= 300
  • words[i] consists of lowercase English letters.

Solution

Python Solution

class Solution:
    def palindromePairs(self, words: List[str]) -> List[List[int]]:
        def is_palindrome(word: str, start: int, end: int) -> bool:
            while start < end:
                if word[start] != word[end]:
                    return False
                start += 1
                end -= 1
            return True
        
        word_dict = {word: i for i, word in enumerate(words)}
        result = []
        
        for i, word in enumerate(words):
            n = len(word)
            for j in range(n + 1):
                # Check prefix
                if is_palindrome(word, j, n - 1):
                    rev_prefix = word[:j][::-1]
                    if rev_prefix in word_dict and word_dict[rev_prefix] != i:
                        result.append([i, word_dict[rev_prefix]])
                
                # Check suffix (skip if j = 0 to avoid duplicates)
                if j > 0 and is_palindrome(word, 0, j - 1):
                    rev_suffix = word[j:][::-1]
                    if rev_suffix in word_dict and word_dict[rev_suffix] != i:
                        result.append([word_dict[rev_suffix], i])
        
        return result

Java Solution

class Solution {
    private boolean isPalindrome(String word, int start, int end) {
        while (start < end) {
            if (word.charAt(start) != word.charAt(end)) {
                return false;
            }
            start++;
            end--;
        }
        return true;
    }
    
    public List> palindromePairs(String[] words) {
        List> result = new ArrayList<>();
        Map wordMap = new HashMap<>();
        
        // Build word map
        for (int i = 0; i < words.length; i++) {
            wordMap.put(words[i], i);
        }
        
        // Check each word
        for (int i = 0; i < words.length; i++) {
            String word = words[i];
            int len = word.length();
            
            for (int j = 0; j <= len; j++) {
                // Check prefix
                if (isPalindrome(word, j, len - 1)) {
                    String revPrefix = new StringBuilder(word.substring(0, j)).reverse().toString();
                    if (wordMap.containsKey(revPrefix) && wordMap.get(revPrefix) != i) {
                        result.add(Arrays.asList(i, wordMap.get(revPrefix)));
                    }
                }
                
                // Check suffix
                if (j > 0 && isPalindrome(word, 0, j - 1)) {
                    String revSuffix = new StringBuilder(word.substring(j)).reverse().toString();
                    if (wordMap.containsKey(revSuffix) && wordMap.get(revSuffix) != i) {
                        result.add(Arrays.asList(wordMap.get(revSuffix), i));
                    }
                }
            }
        }
        
        return result;
    }
}

        vector> result;
        unordered_map wordMap;
        
        // Build word map
        for (int i = 0; i < words.size(); i++) {
            wordMap[words[i]] = i;
        }
        
        // Check each word
        for (int i = 0; i < words.size(); i++) {
            const string& word = words[i];
            int len = word.length();
            
            for (int j = 0; j <= len; j++) {
                // Check prefix
                if (isPalindrome(word, j, len - 1)) {
                    string revPrefix = word.substr(0, j);
                    reverse(revPrefix.begin(), revPrefix.end());
                    auto it = wordMap.find(revPrefix);

                    reverse(revSuffix.begin(), revSuffix.end());
                    auto it = wordMap.find(revSuffix);
                    if (it != wordMap.end() && it->second != i) {
                        result.push_back({it->second, i});
                    }
                }
            }
        }
        
        return result;
    }

var palindromePairs = function(words) {
    const isPalindrome = (word, start, end) => {
        while (start < end) {
            if (word[start] !== word[end]) {
                return false;
            }
            start++;
            end--;
        }
        return true;
    };
    
    const result = [];
    const wordMap = new Map();
    
    // Build word map
    words.forEach((word, i) => {
        wordMap.set(word, i);
    });
    
    // Check each word
    words.forEach((word, i) => {
        const len = word.length;
        
        for (let j = 0; j <= len; j++) {
            // Check prefix
            if (isPalindrome(word, j, len - 1)) {
                const revPrefix = word.slice(0, j).split('').reverse().join('');
                if (wordMap.has(revPrefix) && wordMap.get(revPrefix) !== i) {
                    result.push([i, wordMap.get(revPrefix)]);
                }
            }
            
            // Check suffix
            if (j > 0 && isPalindrome(word, 0, j - 1)) {
                const revSuffix = word.slice(j).split('').reverse().join('');
                if (wordMap.has(revSuffix) && wordMap.get(revSuffix) !== i) {
                    result.push([wordMap.get(revSuffix), i]);
                }
            }
        }
    });
    
    return result;
};
