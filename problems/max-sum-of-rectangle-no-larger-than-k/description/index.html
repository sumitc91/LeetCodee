363. Max Sum of Rectangle No Larger Than K

Jump to Solution: Python Java C++ JavaScript C#

Problem Description

Given an m x n matrix matrix and an integer k, return the max sum of a rectangle in the matrix such that its sum is no larger than k.

It is guaranteed that there will be a rectangle with a sum no larger than k.

Example 1:

Input: matrix = [[1,0,1],[0,-2,3]], k = 2
Output: 2
Explanation: Because the sum of the blue rectangle [[0, 1], [-2, 3]] is 2, and 2 is the max number no larger than k (k = 2).

Example 2:

Input: matrix = [[2,2,-1]], k = 3
Output: 3

Constraints:

  • m == matrix.length
  • n == matrix[i].length
  • 1 <= m, n <= 100
  • -100 <= matrix[i][j] <= 100
  • -10⁵ <= k <= 10⁵

Solution

Python Solution

class Solution:
    def maxSumSubmatrix(self, matrix: List[List[int]], k: int) -> int:
        m, n = len(matrix), len(matrix[0])
        result = float('-inf')
        
        # For each left boundary
        for left in range(n):
            row_sum = [0] * m
            # For each right boundary
            for right in range(left, n):
                # Calculate row sums between left and right
                for i in range(m):
                    row_sum[i] += matrix[i][right]
                
                # Find max subarray sum no more than k
                curr_sum = 0
                max_sum = float('-inf')
                prefix_sums = [0]
                
                for sum_val in row_sum:
                    curr_sum += sum_val
                    # Find the smallest prefix sum that makes curr_sum - prefix >= k
                    target = curr_sum - k
                    idx = bisect_left(prefix_sums, target)
                    if idx < len(prefix_sums):
                        max_sum = max(max_sum, curr_sum - prefix_sums[idx])
                    insort(prefix_sums, curr_sum)
                
                result = max(result, max_sum)
                if result == k:
                    return k
        
        return result

Time Complexity: O(n²m log m)

Where n is the width and m is the height of the matrix.

Space Complexity: O(m)

For storing row sums and prefix sums.

Java Solution

class Solution {
    public int maxSumSubmatrix(int[][] matrix, int k) {
        int m = matrix.length, n = matrix[0].length;
        int result = Integer.MIN_VALUE;
        
        // For each left boundary
        for (int left = 0; left < n; left++) {
            int[] rowSum = new int[m];
            // For each right boundary
            for (int right = left; right < n; right++) {
                // Calculate row sums between left and right
                for (int i = 0; i < m; i++) {
                    rowSum[i] += matrix[i][right];
                }
                
                // Find max subarray sum no more than k
                TreeSet prefixSums = new TreeSet<>();
                prefixSums.add(0);
                int currSum = 0;
                int maxSum = Integer.MIN_VALUE;
                
                for (int sum : rowSum) {
                    currSum += sum;
                    // Find the smallest prefix sum that makes currSum - prefix >= k
                    Integer prefix = prefixSums.ceiling(currSum - k);
                    if (prefix != null) {
                        maxSum = Math.max(maxSum, currSum - prefix);
                    }
                    prefixSums.add(currSum);
                }
                
                result = Math.max(result, maxSum);
                if (result == k) return k;
            }
        }
        
        return result;
    }
}

JavaScript Solution

/**
 * @param {number[][]} matrix
 * @param {number} k
 * @return {number}
 */
var maxSumSubmatrix = function(matrix, k) {
    const m = matrix.length, n = matrix[0].length;
    let result = -Infinity;
    
    // For each left boundary
    for (let left = 0; left < n; left++) {
        const rowSum = new Array(m).fill(0);
        // For each right boundary
        for (let right = left; right < n; right++) {
            // Calculate row sums between left and right
            for (let i = 0; i < m; i++) {
                rowSum[i] += matrix[i][right];
            }
            
            // Find max subarray sum no more than k
            const prefixSums = [0];
            let currSum = 0;
            let maxSum = -Infinity;
            
            for (const sum of rowSum) {
                currSum += sum;
                // Find the smallest prefix sum that makes currSum - prefix >= k
                const target = currSum - k;
                let idx = 0;
                while (idx < prefixSums.length && prefixSums[idx] < target) idx++;
                if (idx < prefixSums.length) {
                    maxSum = Math.max(maxSum, currSum - prefixSums[idx]);
                }
                
                // Insert currSum in sorted order
                idx = 0;
                while (idx < prefixSums.length && prefixSums[idx] < currSum) idx++;
                prefixSums.splice(idx, 0, currSum);
            }
            
            result = Math.max(result, maxSum);
            if (result === k) return k;
        }
    }
    
    return result;
};

public class Solution {
    public int MaxSumSubmatrix(int[][] matrix, int k) {
        int m = matrix.Length, n = matrix[0].Length;
        int result = int.MinValue;
        
        // For each left boundary
        for (int left = 0; left < n; left++) {
            int[] rowSum = new int[m];
            // For each right boundary
            for (int right = left; right < n; right++) {
                // Calculate row sums between left and right
                for (int i = 0; i < m; i++) {
                    rowSum[i] += matrix[i][right];
                }
                
                // Find max subarray sum no more than k
                SortedSet prefixSums = new SortedSet();
                prefixSums.Add(0);
                int currSum = 0;
                int maxSum = int.MinValue;
                
                foreach (int sum in rowSum) {
                    currSum += sum;
                    // Find the smallest prefix sum that makes currSum - prefix >= k
                    var prefix = prefixSums.GetViewBetween(int.MinValue, currSum - k);
                    if (prefix.Count > 0) {
                        maxSum = Math.Max(maxSum, currSum - prefix.Min);
                    }
                    prefixSums.Add(currSum);
                }
                
                result = Math.Max(result, maxSum);
                if (result == k) return k;
            }
        }
        
        return result;
    }
}
