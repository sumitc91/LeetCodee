116. Populating Next Right Pointers in Each Node

Medium

Problem Description

You are given a perfect binary tree where all leaves are on the same level, and every parent has two children. The binary tree has the following definition:

struct Node {
  int val;
  Node *left;
  Node *right;
  Node *next;
}

Populate each next pointer to point to its next right node. If there is no next right node, the next pointer should be set to NULL.

Initially, all next pointers are set to NULL.

Examples

Example 1:
Input: root = [1,2,3,4,5,6,7]
Output: [1,#,2,3,#,4,5,6,7,#]
Explanation: Given the above perfect binary tree, your function should populate each next pointer to point to its next right node. The serialized output is [1,#,2,3,#,4,5,6,7,#] where # represents the NULL pointer.

Example 2:
Input: root = []
Output: []

Python Solution


"""
# Definition for a Node.
class Node:
    def __init__(self, val: int = 0, left: 'Node' = None, right: 'Node' = None, next: 'Node' = None):
        self.val = val
        self.left = left
        self.right = right
        self.next = next
"""
def connect(root: 'Optional[Node]') -> 'Optional[Node]':
    if not root:
        return None
    
    # Start with the first level
    leftmost = root
    
    # Continue until we reach a leaf node
    while leftmost.left:
        current = leftmost
        
        while current:
            # Connect the left child to the right child
            current.left.next = current.right
            
            # Connect the right child to the next node's left child
            if current.next:
                current.right.next = current.next.left
            
            # Move to the next node at the same level
            current = current.next
        
        # Move to the next level
        leftmost = leftmost.left
    
    return root

/*
// Definition for a Node.
class Node {
    public int val;
    public Node left;
    public Node right;
    public Node next;

    public Node() {}
    
    public Node(int _val) {
        val = _val;
    }

    public Node(int _val, Node _left, Node _right, Node _next) {
        val = _val;
        left = _left;
        right = _right;
        next = _next;
    }

class Solution {
    public Node connect(Node root) {
        if (root == null) {
            return null;
        }
        
        // Start with the first level
        Node leftmost = root;
        
        // Continue until we reach a leaf node
        while (leftmost.left != null) {
            Node current = leftmost;
            
            while (current != null) {
                // Connect the left child to the right child
                current.left.next = current.right;
                
                // Connect the right child to the next node's left child
                if (current.next != null) {
                    current.right.next = current.next.left;
                }
                
                // Move to the next node at the same level
                current = current.next;
            }
            
            // Move to the next level
            leftmost = leftmost.left;
        }
        
        return root;
    }
}

    Node(int _val) : val(_val), left(NULL), right(NULL), next(NULL) {}

        : val(_val), left(_left), right(_right), next(_next) {}

class Solution {
public:
    Node* connect(Node* root) {
        if (!root) {
            return nullptr;
        }
        
        // Start with the first level
        Node* leftmost = root;
        
        // Continue until we reach a leaf node
        while (leftmost->left) {
            Node* current = leftmost;
            
            while (current) {
                // Connect the left child to the right child
                current->left->next = current->right;
                
                // Connect the right child to the next node's left child
                if (current->next) {
                    current->right->next = current->next->left;
                }
                
                // Move to the next node at the same level
                current = current->next;
            }
            
            // Move to the next level
            leftmost = leftmost->left;
        }
        
        return root;
    }
};

Complexity Analysis

  • Time Complexity: O(n) where n is the number of nodes in the tree
  • Space Complexity: O(1) as we only use constant extra space

Solution Explanation

This solution uses a level-by-level approach:

  • Key concept:
    • Perfect binary tree
    • Level-wise traversal
    • Next pointer linking
  • Algorithm steps:
    • Start from leftmost node
    • Connect children nodes
    • Link across parent nodes
    • Move to next level

Key points:

  • Constant extra space
  • Level-wise processing
  • Perfect tree property
  • Efficient traversal
