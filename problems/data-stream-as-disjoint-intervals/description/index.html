352. Data Stream as Disjoint Intervals

Jump to Solution: Python Java C++ JavaScript C#

Problem Description

Given a data stream input of non-negative integers a₁, a₂, ..., aₙ, summarize the numbers seen so far as a list of disjoint intervals.

Implement the SummaryRanges class:

  • SummaryRanges() Initializes the object with an empty stream.
  • void addNum(int value) Adds the integer value to the stream.
  • int[][] getIntervals() Returns a summary of the integers in the stream currently as a list of disjoint intervals [starti, endi]. The answer should be sorted by starti.

Example 1:

[[], [1], [], [3], [], [7], [], [2], [], [6], []]

Constraints:

  • 0 <= value <= 10⁴
  • At most 3 * 10⁴ calls will be made to addNum and getIntervals.

Solution

Python Solution

class SummaryRanges:
    def __init__(self):
        self.intervals = []

    def addNum(self, value: int) -> None:
        # Binary search to find insertion position
        left, right = 0, len(self.intervals)
        while left < right:
            mid = (left + right) // 2
            if self.intervals[mid][0] <= value:
                left = mid + 1
            else:
                right = mid
        
        # Insert position found
        pos = left
        
        # Check if can merge with previous interval
        if pos > 0 and self.intervals[pos-1][1] + 1 >= value:
            if value <= self.intervals[pos-1][1]:
                return
            self.intervals[pos-1][1] = value
            # Check if can merge with next interval
            if pos < len(self.intervals) and value + 1 >= self.intervals[pos][0]:
                self.intervals[pos-1][1] = self.intervals[pos][1]
                self.intervals.pop(pos)
            return
        
        # Check if can merge with next interval
        if pos < len(self.intervals) and value + 1 >= self.intervals[pos][0]:
            self.intervals[pos][0] = value
            return
        
        # Insert new interval
        self.intervals.insert(pos, [value, value])

    def getIntervals(self) -> List[List[int]]:
        return self.intervals

Time Complexity:

  • addNum: O(log n) for binary search + O(n) for insertion
  • getIntervals: O(1)

Space Complexity: O(n)

Where n is the number of disjoint intervals.

Java Solution

class SummaryRanges {
    private TreeMap intervals;

    public SummaryRanges() {
        intervals = new TreeMap<>();
    }
    
    public void addNum(int value) {
        // Find the interval that starts after value
        Map.Entry ceiling = intervals.ceilingEntry(value + 1);
        // Find the interval that starts before or at value
        Map.Entry floor = intervals.floorEntry(value);
        
        if (floor != null && floor.getKey() <= value && value <= floor.getValue()) {
            return; // value is already covered
        }
        
        boolean mergePrev = (floor != null && floor.getValue() + 1 >= value);
        boolean mergeNext = (ceiling != null && value + 1 >= ceiling.getKey());
        
        if (mergePrev && mergeNext) {
            // Merge both intervals
            intervals.put(floor.getKey(), ceiling.getValue());
            intervals.remove(ceiling.getKey());
        } else if (mergePrev) {
            // Extend previous interval
            intervals.put(floor.getKey(), Math.max(value, floor.getValue()));
        } else if (mergeNext) {
            // Extend next interval
            intervals.put(value, ceiling.getValue());
            intervals.remove(ceiling.getKey());
        } else {
            // Create new interval
            intervals.put(value, value);
        }
    }
    
    public int[][] getIntervals() {
        int[][] result = new int[intervals.size()][2];
        int i = 0;
        for (Map.Entry entry : intervals.entrySet()) {
            result[i][0] = entry.getKey();
            result[i][1] = entry.getValue();
            i++;
        }
        return result;
    }
}

C++ Solution

class SummaryRanges {
private:
    set> intervals;
    
public:
    SummaryRanges() {}
    
    void addNum(int value) {
        auto it = intervals.lower_bound({value, value});
        int start = value, end = value;
        
        // Check if can merge with previous interval
        if (it != intervals.begin()) {
            auto prev = prev(it);
            if (prev->second + 1 >= value) {
                start = prev->first;
                end = max(value, prev->second);
                intervals.erase(prev);
            }
        }
        
        // Check if can merge with next intervals
        while (it != intervals.end() && end + 1 >= it->first) {
            end = max(end, it->second);
            it = intervals.erase(it);
        }
        
        intervals.insert({start, end});
    }
    
    vector> getIntervals() {
        vector> result;
        for (const auto& interval : intervals) {
            result.push_back({interval.first, interval.second});
        }
        return result;
    }
};

JavaScript Solution

class SummaryRanges {
    constructor() {
        this.intervals = [];
    }
    
    /** 
     * @param {number} value
     * @return {void}
     */
    addNum(value) {
        // Binary search to find insertion position
        let left = 0, right = this.intervals.length;
        while (left < right) {
            const mid = Math.floor((left + right) / 2);
            if (this.intervals[mid][0] <= value) {
                left = mid + 1;
            } else {
                right = mid;
            }
        }
        
        const pos = left;
        
        // Check if can merge with previous interval
        if (pos > 0 && this.intervals[pos-1][1] + 1 >= value) {
            if (value <= this.intervals[pos-1][1]) return;
            this.intervals[pos-1][1] = value;
            // Check if can merge with next interval
            if (pos < this.intervals.length && value + 1 >= this.intervals[pos][0]) {
                this.intervals[pos-1][1] = this.intervals[pos][1];
                this.intervals.splice(pos, 1);
            }
            return;
        }
        
        // Check if can merge with next interval
        if (pos < this.intervals.length && value + 1 >= this.intervals[pos][0]) {
            this.intervals[pos][0] = value;
            return;
        }
        
        // Insert new interval
        this.intervals.splice(pos, 0, [value, value]);
    }
    
    /**
     * @return {number[][]}
     */
    getIntervals() {
        return this.intervals;
    }
}

C# Solution

public class SummaryRanges {
    private SortedDictionary intervals;

    public SummaryRanges() {
        intervals = new SortedDictionary();
    }
    
    public void AddNum(int value) {
        // Find the interval that starts after value
        var ceiling = intervals.Where(x => x.Key > value).FirstOrDefault();
        // Find the interval that starts before or at value
        var floor = intervals.Where(x => x.Key <= value).LastOrDefault();
        
        if (floor.Key != 0 && floor.Key <= value && value <= floor.Value) {
            return; // value is already covered
        }
        
        bool mergePrev = (floor.Key != 0 && floor.Value + 1 >= value);
        bool mergeNext = (ceiling.Key != 0 && value + 1 >= ceiling.Key);
        
        if (mergePrev && mergeNext) {
            // Merge both intervals
            intervals[floor.Key] = ceiling.Value;
            intervals.Remove(ceiling.Key);
        } else if (mergePrev) {
            // Extend previous interval
            intervals[floor.Key] = Math.Max(value, floor.Value);
        } else if (mergeNext) {
            // Extend next interval
            intervals.Add(value, ceiling.Value);
            intervals.Remove(ceiling.Key);
        } else {
            // Create new interval
            intervals.Add(value, value);
        }
    }
    
    public int[][] GetIntervals() {
        return intervals.Select(x => new int[] { x.Key, x.Value }).ToArray();
    }
}
