472. Concatenated Words

Jump to Solution: Python Java C++ JavaScript C#

Problem Description

Given an array of strings words (without duplicates), return all the concatenated words in the given list of words.

A concatenated word is defined as a string that is comprised entirely of at least two shorter words in the given array.

Example 1:

Input: words = ["cat","cats","catsdogcats","dog","dogcatsdog","hippopotamuses","rat","ratcatdogcat"]
Output: ["catsdogcats","dogcatsdog","ratcatdogcat"]
Explanation: "catsdogcats" can be concatenated by "cats", "dog" and "cats"; 
"dogcatsdog" can be concatenated by "dog", "cats" and "dog"; 
"ratcatdogcat" can be concatenated by "rat", "cat", "dog" and "cat".

Example 2:

Input: words = ["cat","dog","catdog"]
Output: ["catdog"]

Constraints:

  • 1 <= words.length <= 10⁴
  • 1 <= words[i].length <= 30
  • words[i] consists of only lowercase English letters.
  • All the strings of words are unique.
  • 1 <= sum(words[i].length) <= 10⁵

Solution

Python Solution

class Solution:
    def findAllConcatenatedWordsInADict(self, words: List[str]) -> List[str]:
        word_set = set(words)
        result = []
        
        def can_form(word, start, count):
            if start == len(word) and count >= 2:
                return True
            
            for i in range(start + 1, len(word) + 1):
                curr = word[start:i]
                if curr in word_set:
                    if can_form(word, i, count + 1):
                        return True
            return False
        
        for word in words:
            word_set.remove(word)
            if word and can_form(word, 0, 0):
                result.append(word)
            word_set.add(word)
        
        return result

Where n is the number of words and l is the maximum length of a word.

Java Solution

class Solution {
    private Set wordSet;
    
    public List findAllConcatenatedWordsInADict(String[] words) {
        wordSet = new HashSet<>(Arrays.asList(words));
        List result = new ArrayList<>();
        
        for (String word : words) {
            wordSet.remove(word);
            if (word.length() > 0 && canForm(word, 0, 0)) {
                result.add(word);
            }
            wordSet.add(word);
        }
        
        return result;
    }
    
    private boolean canForm(String word, int start, int count) {
        if (start == word.length() && count >= 2) {
            return true;
        }
        
        for (int i = start + 1; i <= word.length(); i++) {
            String curr = word.substring(start, i);
            if (wordSet.contains(curr)) {
                if (canForm(word, i, count + 1)) {
                    return true;
                }
            }
        }
        return false;
    }
}

C++ Solution

class Solution {
private:
    unordered_set wordSet;
    
    bool canForm(const string& word, int start, int count) {
        if (start == word.length() && count >= 2) {
            return true;
        }
        
        for (int i = start + 1; i <= word.length(); i++) {
            string curr = word.substr(start, i - start);
            if (wordSet.count(curr)) {
                if (canForm(word, i, count + 1)) {
                    return true;
                }
            }
        }
        return false;
    }
    
public:
    vector findAllConcatenatedWordsInADict(vector& words) {
        wordSet = unordered_set(words.begin(), words.end());
        vector result;
        
        for (const string& word : words) {
            wordSet.erase(word);
            if (word.length() > 0 && canForm(word, 0, 0)) {
                result.push_back(word);
            }
            wordSet.insert(word);
        }
        
        return result;
    }
};

 * @param {string[]} words

var findAllConcatenatedWordsInADict = function(words) {
    const wordSet = new Set(words);
    const result = [];
    
    const canForm = (word, start, count) => {
        if (start === word.length && count >= 2) {
            return true;
        }
        
        for (let i = start + 1; i <= word.length; i++) {
            const curr = word.slice(start, i);
            if (wordSet.has(curr)) {
                if (canForm(word, i, count + 1)) {
                    return true;
                }
            }
        }
        return false;
    };
    
    for (const word of words) {
        wordSet.delete(word);
        if (word.length > 0 && canForm(word, 0, 0)) {
            result.push(word);
        }
        wordSet.add(word);
    }
    
    return result;
};

public class Solution {
    private HashSet wordSet;
    
    public IList FindAllConcatenatedWordsInADict(string[] words) {
        wordSet = new HashSet(words);
        IList result = new List();
        
        foreach (string word in words) {
            wordSet.Remove(word);
            if (word.Length > 0 && CanForm(word, 0, 0)) {
                result.Add(word);
            }
            wordSet.Add(word);
        }
        
        return result;
    }
    
    private bool CanForm(string word, int start, int count) {
        if (start == word.Length && count >= 2) {
            return true;
        }
        
        for (int i = start + 1; i <= word.Length; i++) {
            string curr = word.Substring(start, i - start);
            if (wordSet.Contains(curr)) {
                if (CanForm(word, i, count + 1)) {
                    return true;
                }
            }
        }
        return false;
    }
}
