525. Contiguous Array

Jump to Solution: Python Java C++ JavaScript C#

Problem Description

Given a binary array nums, return the maximum length of a contiguous subarray with an equal number of 0 and 1.

Example 1:

Input: nums = [0,1]
Output: 2
Explanation: [0, 1] is the longest contiguous subarray with an equal number of 0 and 1.

Example 2:

Input: nums = [0,1,0]
Output: 2
Explanation: [0, 1] (or [1, 0]) is a longest contiguous subarray with equal number of 0 and 1.

Constraints:

  • 1 <= nums.length <= 10⁵
  • nums[i] is either 0 or 1

Solution

Python Solution

class Solution:
    def findMaxLength(self, nums: List[int]) -> int:
        # Map to store count and its first occurrence index
        count_map = {0: -1}
        count = max_length = 0
        
        for i, num in enumerate(nums):
            # Increment count for 1, decrement for 0
            count += 1 if num == 1 else -1
            
            # If count seen before, update max_length
            if count in count_map:
                max_length = max(max_length, i - count_map[count])
            else:
                count_map[count] = i
        
        return max_length

Time Complexity: O(n)

Where n is the length of the input array. We traverse the array once.

Space Complexity: O(n)

For storing counts in the hash map.

Java Solution

class Solution {
    public int findMaxLength(int[] nums) {
        // Map to store count and its first occurrence index
        Map countMap = new HashMap<>();
        countMap.put(0, -1);
        int count = 0, maxLength = 0;
        
        for (int i = 0; i < nums.length; i++) {
            // Increment count for 1, decrement for 0
            count += nums[i] == 1 ? 1 : -1;
            
            // If count seen before, update maxLength
            if (countMap.containsKey(count)) {
                maxLength = Math.max(maxLength, i - countMap.get(count));
            } else {
                countMap.put(count, i);
            }
        }
        
        return maxLength;
    }
}

        // Map to store count and its first occurrence index
        Dictionary countMap = new Dictionary();
        countMap[0] = -1;
        int count = 0, maxLength = 0;
        
        for (int i = 0; i < nums.Length; i++) {

            count += nums[i] == 1 ? 1 : -1;
            
            // If count seen before, update maxLength
            if (countMap.ContainsKey(count)) {
                maxLength = Math.Max(maxLength, i - countMap[count]);
            } else {
                countMap[count] = i;
            }
        }
        
        return maxLength;
    }
}
