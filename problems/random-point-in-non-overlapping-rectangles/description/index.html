497. Random Point in Non-overlapping Rectangles

Jump to Solution: Python Java C++ JavaScript C#

Problem Description

You are given an array of non-overlapping axis-aligned rectangles rects where rects[i] = [ai, bi, xi, yi] indicates that (ai, bi) is the bottom-left corner point of the ith rectangle and (xi, yi) is the top-right corner point of the ith rectangle. Design an algorithm to pick a random integer point inside the space covered by the rectangles.

A point on the perimeter of a rectangle is included in the space covered by the rectangles. All integer points inside the space should be equally likely to be returned.

Implement the Solution class:

  • Solution(int[][] rects) Initializes the object with the array of rectangles rects.
  • int[] pick() Returns a random integer point [u, v] inside the space covered by the rectangles.

Example 1:

Constraints:

  • 1 <= rects.length <= 100
  • rects[i].length == 4
  • -10⁹ <= ai < xi <= 10⁹
  • -10⁹ <= bi < yi <= 10⁹
  • xi - ai <= 2000
  • yi - bi <= 2000
  • All the rectangles do not overlap.
  • At most 10⁴ calls will be made to pick.

Solution

Python Solution

class Solution:
    def __init__(self, rects: List[List[int]]):
        self.rects = rects
        self.total_points = 0
        self.rect_weights = []
        
        # Calculate total points and weights for each rectangle
        for rect in rects:
            points = (rect[2] - rect[0] + 1) * (rect[3] - rect[1] + 1)
            self.total_points += points
            self.rect_weights.append(self.total_points)
    
    def pick(self) -> List[int]:
        # Pick a random point index
        point = random.randint(1, self.total_points)
        
        # Find the rectangle that contains this point
        rect_idx = bisect.bisect_left(self.rect_weights, point)
        rect = self.rects[rect_idx]
        
        # Calculate previous points to get local point index
        prev = self.rect_weights[rect_idx - 1] if rect_idx > 0 else 0
        point = point - prev
        
        # Calculate width and height of current rectangle
        width = rect[2] - rect[0] + 1
        height = rect[3] - rect[1] + 1
        
        # Convert point index to x,y coordinates
        x = rect[0] + (point - 1) % width
        y = rect[1] + (point - 1) // width
        
        return [x, y]

  • Constructor: O(n) where n is the number of rectangles
  • Pick: O(log n) for binary search

Java Solution

class Solution {
    private int[][] rects;
    private int totalPoints;
    private TreeMap rectMap;
    private Random rand;
    
    public Solution(int[][] rects) {
        this.rects = rects;
        this.totalPoints = 0;
        this.rectMap = new TreeMap<>();
        this.rand = new Random();
        
        // Calculate total points and weights for each rectangle
        for (int i = 0; i < rects.length; i++) {
            int[] rect = rects[i];
            int points = (rect[2] - rect[0] + 1) * (rect[3] - rect[1] + 1);
            totalPoints += points;
            rectMap.put(totalPoints, i);
        }
    }
    
    public int[] pick() {
        // Pick a random point index
        int point = rand.nextInt(totalPoints) + 1;
        
        // Find the rectangle that contains this point
        int rectIdx = rectMap.ceilingEntry(point).getValue();
        int[] rect = rects[rectIdx];
        
        // Calculate previous points to get local point index
        int prev = rectIdx > 0 ? rectMap.floorKey(point - 1) : 0;
        point = point - prev;
        
        // Calculate width and height of current rectangle
        int width = rect[2] - rect[0] + 1;
        int height = rect[3] - rect[1] + 1;
        
        // Convert point index to x,y coordinates
        int x = rect[0] + (point - 1) % width;
        int y = rect[1] + (point - 1) / width;
        
        return new int[]{x, y};
    }
}

/**

class Solution {
    constructor(rects) {
        this.rects = rects;
        this.totalPoints = 0;
        this.rectWeights = [];
        
        // Calculate total points and weights for each rectangle
        for (const rect of rects) {
            const points = (rect[2] - rect[0] + 1) * (rect[3] - rect[1] + 1);
            this.totalPoints += points;
            this.rectWeights.push(this.totalPoints);
        }
    }
    
    /**
     * @return {number[]}
     */
    pick() {
        // Pick a random point index
        const point = Math.floor(Math.random() * this.totalPoints) + 1;
        
        // Find the rectangle that contains this point
        const rectIdx = this.rectWeights.findIndex(w => w >= point);
        const rect = this.rects[rectIdx];
        
        // Calculate previous points to get local point index
        const prev = rectIdx > 0 ? this.rectWeights[rectIdx - 1] : 0;
        const localPoint = point - prev;
        
        // Calculate width and height of current rectangle
        const width = rect[2] - rect[0] + 1;
        const height = rect[3] - rect[1] + 1;
        
        // Convert point index to x,y coordinates
        const x = rect[0] + ((localPoint - 1) % width);
        const y = rect[1] + Math.floor((localPoint - 1) / width);
        
        return [x, y];
    }
}

C# Solution

public class Solution {
    private int[][] rects;
    private int totalPoints;
    private int[] rectWeights;
    private Random rand;
    
    public Solution(int[][] rects) {
        this.rects = rects;
        this.totalPoints = 0;
        this.rectWeights = new int[rects.Length];
        this.rand = new Random();
        
        // Calculate total points and weights for each rectangle
        for (int i = 0; i < rects.Length; i++) {
            int points = (rects[i][2] - rects[i][0] + 1) * (rects[i][3] - rects[i][1] + 1);
            totalPoints += points;
            rectWeights[i] = totalPoints;
        }
    }
    
    public int[] Pick() {
        // Pick a random point index
        int point = rand.Next(totalPoints) + 1;
        
        // Find the rectangle that contains this point
        int rectIdx = Array.BinarySearch(rectWeights, point);
        if (rectIdx < 0) rectIdx = ~rectIdx;
        int[] rect = rects[rectIdx];
        
        // Calculate previous points to get local point index
        int prev = rectIdx > 0 ? rectWeights[rectIdx - 1] : 0;
        point = point - prev;
        
        // Calculate width and height of current rectangle
        int width = rect[2] - rect[0] + 1;
        int height = rect[3] - rect[1] + 1;
        
        // Convert point index to x,y coordinates
        int x = rect[0] + (point - 1) % width;
        int y = rect[1] + (point - 1) / width;
        
        return new int[] { x, y };
    }
}
