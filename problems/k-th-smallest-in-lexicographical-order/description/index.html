440. K-th Smallest in Lexicographical Order

Jump to Solution: Python Java C++ JavaScript C#

Problem Description

Given two integers n and k, return the kth lexicographically smallest integer in the range [1, n].

Example 1:

Input: n = 13, k = 2
Output: 10
Explanation: The lexicographical order is [1, 10, 11, 12, 13, 2, 3, 4, 5, 6, 7, 8, 9], so the second smallest number is 10.

Example 2:

Input: n = 1, k = 1
Output: 1

Constraints:

  • 1 <= k <= n <= 10⁹

Solution

Python Solution

class Solution:
    def findKthNumber(self, n: int, k: int) -> int:
        def count_prefix(prefix, n):
            curr = prefix
            next_prefix = prefix + 1
            count = 0
            
            while curr <= n:
                count += min(n + 1, next_prefix) - curr
                curr *= 10
                next_prefix *= 10
            
            return count
        
        curr = 1
        k -= 1  # Convert to 0-based index
        
        while k > 0:
            count = count_prefix(curr, n)
            if k >= count:
                # Skip this prefix
                k -= count
                curr += 1
            else:
                # Go down this prefix
                k -= 1
                curr *= 10
        
        return curr

Time Complexity: O(log n * log n)

For each digit position, we need to calculate the count of numbers with that prefix.

Space Complexity: O(1)

Only using a constant amount of extra space.

class Solution {
    public int findKthNumber(int n, int k) {
        int curr = 1;
        k = k - 1;  // Convert to 0-based index
        
        while (k > 0) {
            int count = countPrefix(curr, n);
            if (k >= count) {
                // Skip this prefix
                k -= count;
                curr++;
            } else {
                // Go down this prefix
                k--;
                curr *= 10;
            }
        }
        
        return curr;
    }
    
    private int countPrefix(long prefix, int n) {
        long curr = prefix;
        long next = prefix + 1;
        int count = 0;
        
        while (curr <= n) {
            count += (int) Math.min(n + 1L, next) - curr;
            curr *= 10;
            next *= 10;
        }
        
        return count;
    }
}
