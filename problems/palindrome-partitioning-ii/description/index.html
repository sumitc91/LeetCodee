132. Palindrome Partitioning II

Hard

Problem Description

Given a string s, partition s such that every substring of the partition is a palindrome.

Return the minimum cuts needed for a palindrome partitioning of s.

Examples

Example 1:
Input: s = "aab"
Output: 1
Explanation: The palindrome partitioning ["aa","b"] could be produced using 1 cut.

Example 2:
Input: s = "a"
Output: 0

Example 3:
Input: s = "ab"
Output: 1

Python Solution


def minCut(s: str) -> int:
    n = len(s)
    # dp[i] represents the minimum cuts needed for s[0:i+1]
    dp = [0] * n
    # is_palindrome[i][j] represents whether s[i:j+1] is palindrome
    is_palindrome = [[False] * n for _ in range(n)]
    
    # Initialize palindrome table
    for i in range(n):
        is_palindrome[i][i] = True
    
    # Fill palindrome table
    for length in range(2, n + 1):
        for start in range(n - length + 1):
            end = start + length - 1
            if length == 2:
                is_palindrome[start][end] = s[start] == s[end]
            else:
                is_palindrome[start][end] = s[start] == s[end] and is_palindrome[start + 1][end - 1]
    
    # Fill dp table
    for i in range(n):
        if is_palindrome[0][i]:
            dp[i] = 0
        else:
            dp[i] = i
            for j in range(i):
                if is_palindrome[j + 1][i]:
                    dp[i] = min(dp[i], dp[j] + 1)
    
    return dp[n - 1]

Java Solution


class Solution {
    public int minCut(String s) {
        int n = s.length();
        // dp[i] represents the minimum cuts needed for s[0:i+1]
        int[] dp = new int[n];
        // isPalindrome[i][j] represents whether s[i:j+1] is palindrome
        boolean[][] isPalindrome = new boolean[n][n];
        
        // Initialize palindrome table
        for (int i = 0; i < n; i++) {
            isPalindrome[i][i] = true;
        }
        
        // Fill palindrome table
        for (int length = 2; length <= n; length++) {
            for (int start = 0; start <= n - length; start++) {
                int end = start + length - 1;
                if (length == 2) {
                    isPalindrome[start][end] = s.charAt(start) == s.charAt(end);
                } else {
                    isPalindrome[start][end] = s.charAt(start) == s.charAt(end) && 
                                             isPalindrome[start + 1][end - 1];
                }
            }
        }
        
        // Fill dp table
        for (int i = 0; i < n; i++) {
            if (isPalindrome[0][i]) {
                dp[i] = 0;
            } else {
                dp[i] = i;
                for (int j = 0; j < i; j++) {
                    if (isPalindrome[j + 1][i]) {
                        dp[i] = Math.min(dp[i], dp[j] + 1);
                    }
                }
            }
        }
        
        return dp[n - 1];
    }
}

  • Time Complexity: O(n²) where n is the length of the string
  • Space Complexity: O(n²) for the palindrome table
