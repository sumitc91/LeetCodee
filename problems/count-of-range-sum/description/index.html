327. Count of Range Sum

Jump to Solution: Python Java C++ JavaScript C#

Problem Description

Given an integer array nums and two integers lower and upper, return the number of range sums that lie in [lower, upper] inclusive.

Range sum S(i, j) is defined as the sum of the elements in nums between indices i and j inclusive where i <= j.

Example 1:

Input: nums = [-2,5,-1], lower = -2, upper = 2
Output: 3
Explanation: The three ranges are: [0,0], [2,2], and [0,2] and their respective sums are: -2, -1, 2.

Example 2:

Input: nums = [0], lower = 0, upper = 0
Output: 1

Constraints:

  • 1 <= nums.length <= 10⁵
  • -2³¹ <= nums[i] <= 2³¹ - 1
  • -10⁵ <= lower <= upper <= 10⁵
  • The answer is guaranteed to fit in a 32-bit integer.

Solution

Python Solution

class Solution:
    def countRangeSum(self, nums: List[int], lower: int, upper: int) -> int:
        def mergeSort(start, end):
            if start >= end:
                return 0
                
            mid = (start + end) // 2
            count = mergeSort(start, mid) + mergeSort(mid + 1, end)
            
            # Count pairs
            i = j = mid + 1
            for left in prefix_sum[start:mid + 1]:
                while i <= end and prefix_sum[i] - left < lower:
                    i += 1
                while j <= end and prefix_sum[j] - left <= upper:
                    j += 1
                count += j - i
            
            # Merge
            temp = []
            l, r = start, mid + 1
            while l <= mid and r <= end:
                if prefix_sum[l] <= prefix_sum[r]:
                    temp.append(prefix_sum[l])
                    l += 1
                else:
                    temp.append(prefix_sum[r])
                    r += 1
            temp.extend(prefix_sum[l:mid + 1])
            temp.extend(prefix_sum[r:end + 1])
            
            for i in range(len(temp)):
                prefix_sum[start + i] = temp[i]
                
            return count
        
        # Calculate prefix sum
        prefix_sum = [0]
        for num in nums:
            prefix_sum.append(prefix_sum[-1] + num)
            
        return mergeSort(0, len(prefix_sum) - 1)

Java Solution

class Solution {
    private long[] prefixSum;
    private long[] temp;
    private int lower, upper;
    
    public int countRangeSum(int[] nums, int lower, int upper) {
        this.lower = lower;
        this.upper = upper;
        int n = nums.length;
        prefixSum = new long[n + 1];
        temp = new long[n + 1];
        
        // Calculate prefix sum
        for (int i = 0; i < n; i++) {
            prefixSum[i + 1] = prefixSum[i] + nums[i];
        }
        
        return mergeSort(0, n);
    }
    
    private int mergeSort(int start, int end) {
        if (start >= end) return 0;
        
        int mid = (start + end) / 2;
        int count = mergeSort(start, mid) + mergeSort(mid + 1, end);
        
        // Count pairs
        int i = mid + 1, j = mid + 1;
        for (int left = start; left <= mid; left++) {
            while (i <= end && prefixSum[i] - prefixSum[left] < lower) i++;
            while (j <= end && prefixSum[j] - prefixSum[left] <= upper) j++;
            count += j - i;
        }
        
        // Merge
        merge(start, mid, end);
        return count;
    }
    
    private void merge(int start, int mid, int end) {
        for (int i = start; i <= end; i++) {
            temp[i] = prefixSum[i];
        }
        
        int i = start, j = mid + 1, k = start;
        while (i <= mid && j <= end) {
            if (temp[i] <= temp[j]) {
                prefixSum[k++] = temp[i++];
            } else {
                prefixSum[k++] = temp[j++];
            }
        }
        
        while (i <= mid) {
            prefixSum[k++] = temp[i++];
        }
    }
}

JavaScript Solution

/**
 * @param {number[]} nums
 * @param {number} lower
 * @param {number} upper
 * @return {number}
 */
var countRangeSum = function(nums, lower, upper) {
    const prefixSum = [0];
    for (let i = 0; i < nums.length; i++) {
        prefixSum[i + 1] = prefixSum[i] + nums[i];
    }
    
    function mergeSort(start, end) {
        if (start >= end) return 0;
        
        const mid = Math.floor((start + end) / 2);
        let count = mergeSort(start, mid) + mergeSort(mid + 1, end);
        
        // Count pairs
        let i = mid + 1, j = mid + 1;
        for (let left = start; left <= mid; left++) {
            while (i <= end && prefixSum[i] - prefixSum[left] < lower) i++;
            while (j <= end && prefixSum[j] - prefixSum[left] <= upper) j++;
            count += j - i;
        }
        
        // Merge
        const temp = [];
        let l = start, r = mid + 1;
        while (l <= mid && r <= end) {
            if (prefixSum[l] <= prefixSum[r]) {
                temp.push(prefixSum[l++]);
            } else {
                temp.push(prefixSum[r++]);
            }
        }
        while (l <= mid) temp.push(prefixSum[l++]);
        while (r <= end) temp.push(prefixSum[r++]);
        
        for (let i = 0; i < temp.length; i++) {
            prefixSum[start + i] = temp[i];
        }
        
        return count;
    }
    
    return mergeSort(0, prefixSum.length - 1);
};

public class Solution {
    private long[] prefixSum;
    private long[] temp;
    private int lower, upper;
    
    public int CountRangeSum(int[] nums, int lower, int upper) {
        this.lower = lower;
        this.upper = upper;
        int n = nums.Length;
        prefixSum = new long[n + 1];
        temp = new long[n + 1];
        
        // Calculate prefix sum
        for (int i = 0; i < n; i++) {
            prefixSum[i + 1] = prefixSum[i] + nums[i];
        }
        
        return MergeSort(0, n);
    }
    
    private int MergeSort(int start, int end) {
        if (start >= end) return 0;
        
        int mid = (start + end) / 2;
        int count = MergeSort(start, mid) + MergeSort(mid + 1, end);
        
        // Count pairs
        int i = mid + 1, j = mid + 1;
        for (int left = start; left <= mid; left++) {
            while (i <= end && prefixSum[i] - prefixSum[left] < lower) i++;
            while (j <= end && prefixSum[j] - prefixSum[left] <= upper) j++;
            count += j - i;
        }
        
        // Merge
        Merge(start, mid, end);
        return count;
    }
    
    private void Merge(int start, int mid, int end) {
        for (int i = start; i <= end; i++) {
            temp[i] = prefixSum[i];
        }
        
        int i = start, j = mid + 1, k = start;
        while (i <= mid && j <= end) {
            if (temp[i] <= temp[j]) {
                prefixSum[k++] = temp[i++];
            } else {
                prefixSum[k++] = temp[j++];
            }
        }
        
        while (i <= mid) {
            prefixSum[k++] = temp[i++];
        }
    }
}
