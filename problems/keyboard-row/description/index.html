500. Keyboard Row

Jump to Solution: Python Java C++ JavaScript C#

Problem Description

Given an array of strings words, return the words that can be typed using letters of the alphabet on only one row of American keyboard like the image below.

In the American keyboard:

  • the first row consists of the characters "qwertyuiop",
  • the second row consists of the characters "asdfghjkl", and
  • the third row consists of the characters "zxcvbnm".

Example 1:

Input: words = ["Hello","Alaska","Dad","Peace"]
Output: ["Alaska","Dad"]
Explanation: "Hello" cannot be typed using one row of the keyboard because the letter 'l' is on the second row and 'o' is on the first row.

Example 2:

Input: words = ["omk"]
Output: []
Explanation: "omk" cannot be typed using one row of the keyboard.

Example 3:

Input: words = ["adsdf","sfd"]
Output: ["adsdf","sfd"]
Explanation: All letters in both words can be typed using the second row of the keyboard.

Constraints:

  • 1 <= words.length <= 20
  • 1 <= words[i].length <= 100
  • words[i] consists of English letters (both lowercase and uppercase).

Solution

Python Solution

class Solution:
    def findWords(self, words: List[str]) -> List[str]:
        row1 = set('qwertyuiop')
        row2 = set('asdfghjkl')
        row3 = set('zxcvbnm')
        
        result = []
        for word in words:
            word_set = set(word.lower())
            if word_set <= row1 or word_set <= row2 or word_set <= row3:
                result.append(word)
        
        return result

Time Complexity: O(n × k)

Where n is the number of words and k is the average length of each word.

Space Complexity: O(1)

The keyboard rows are constant size sets.

Java Solution

class Solution {
    public String[] findWords(String[] words) {
        String row1 = "qwertyuiop";
        String row2 = "asdfghjkl";
        String row3 = "zxcvbnm";
        
        List result = new ArrayList<>();
        
        for (String word : words) {
            String lower = word.toLowerCase();
            boolean inRow1 = true, inRow2 = true, inRow3 = true;
            
            for (char c : lower.toCharArray()) {
                if (row1.indexOf(c) == -1) inRow1 = false;
                if (row2.indexOf(c) == -1) inRow2 = false;
                if (row3.indexOf(c) == -1) inRow3 = false;
            }
            
            if (inRow1 || inRow2 || inRow3) {
                result.add(word);
            }
        }
        
        return result.toArray(new String[0]);
    }
}

C++ Solution

class Solution {
public:
    vector findWords(vector& words) {
        unordered_set row1{'q','w','e','r','t','y','u','i','o','p'};
        unordered_set row2{'a','s','d','f','g','h','j','k','l'};
        unordered_set row3{'z','x','c','v','b','n','m'};
        
        vector result;
        
        for (const string& word : words) {
            bool inRow1 = true, inRow2 = true, inRow3 = true;
            
            for (char c : word) {
                c = tolower(c);
                if (row1.find(c) == row1.end()) inRow1 = false;
                if (row2.find(c) == row2.end()) inRow2 = false;
                if (row3.find(c) == row3.end()) inRow3 = false;
            }
            
            if (inRow1 || inRow2 || inRow3) {
                result.push_back(word);
            }
        }
        
        return result;
    }
};

JavaScript Solution

/**
 * @param {string[]} words
 * @return {string[]}
 */
var findWords = function(words) {
    const row1 = new Set('qwertyuiop');
    const row2 = new Set('asdfghjkl');
    const row3 = new Set('zxcvbnm');
    
    return words.filter(word => {
        const chars = new Set(word.toLowerCase());
        return (
            [...chars].every(c => row1.has(c)) ||
            [...chars].every(c => row2.has(c)) ||
            [...chars].every(c => row3.has(c))
        );
    });
};

C# Solution

public class Solution {
    public string[] FindWords(string[] words) {
        HashSet row1 = new HashSet("qwertyuiop");
        HashSet row2 = new HashSet("asdfghjkl");
        HashSet row3 = new HashSet("zxcvbnm");
        
        List result = new List();
        
        foreach (string word in words) {
            HashSet chars = new HashSet(word.ToLower());
            
            if (chars.IsSubsetOf(row1) || chars.IsSubsetOf(row2) || chars.IsSubsetOf(row3)) {
                result.Add(word);
            }
        }
        
        return result.ToArray();
    }
}

Where n is the number of words and k is the average length of each word.
