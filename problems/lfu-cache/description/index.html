460. LFU Cache

Jump to Solution: Python Java C++ JavaScript C#

Problem Description

Design and implement a data structure for a Least Frequently Used (LFU) cache.

Implement the LFUCache class:

  • LFUCache(int capacity) Initializes the object with the capacity of the data structure.
  • int get(int key) Gets the value of the key if it exists in the cache. Otherwise, returns -1.
  • void put(int key, int value) Update the value of the key if present, or inserts the key if not present. When the cache reaches its capacity, it should invalidate and remove the least frequently used key before inserting a new item.

For the purpose of this problem, when there is a tie (i.e., two or more keys with the same frequency), the least recently used key would be invalidated.

Example 1:

Input:
["LFUCache", "put", "put", "get", "put", "get", "get", "put", "get", "get", "get"]
[[2], [1, 1], [2, 2], [1], [3, 3], [2], [3], [4, 4], [1], [3], [4]]
Output:
[null, null, null, 1, null, -1, 3, null, -1, 3, 4]

Explanation:
// cnt(x) = the use counter for key x
// cache=[] will show the last used order for tiebreakers (leftmost element is most recent)
LFUCache lfu = new LFUCache(2);
lfu.put(1, 1);   // cache=[1,_], cnt(1)=1
lfu.put(2, 2);   // cache=[2,1], cnt(2)=1, cnt(1)=1
lfu.get(1);      // return 1
                 // cache=[1,2], cnt(2)=1, cnt(1)=2
lfu.put(3, 3);   // 2 is the LFU key because cnt(2)=1 is the smallest, invalidate 2.
                 // cache=[3,1], cnt(3)=1, cnt(1)=2
lfu.get(2);      // return -1 (not found)
lfu.get(3);      // return 3
                 // cache=[3,1], cnt(3)=2, cnt(1)=2
lfu.put(4, 4);   // Both 1 and 3 have cnt=2, but 1 is LRU, invalidate 1.
                 // cache=[4,3], cnt(4)=1, cnt(3)=2
lfu.get(1);      // return -1 (not found)
lfu.get(3);      // return 3
                 // cache=[3,4], cnt(4)=1, cnt(3)=3
lfu.get(4);      // return 4
                 // cache=[4,3], cnt(4)=2, cnt(3)=3

Constraints:

  • 0 <= capacity <= 10⁴
  • 0 <= key <= 10⁵
  • 0 <= value <= 10⁹
  • At most 2 * 10⁵ calls will be made to get and put.

Solution

Python Solution

class Node:
    def __init__(self, key=None, value=None):
        self.key = key
        self.value = value
        self.freq = 1
        self.prev = None
        self.next = None

class DLinkedList:
    def __init__(self):
        self.head = Node()
        self.tail = Node()
        self.head.next = self.tail
        self.tail.prev = self.head
        self.size = 0
    
    def add_node(self, node):
        node.prev = self.head
        node.next = self.head.next
        self.head.next.prev = node
        self.head.next = node
        self.size += 1
    
    def remove_node(self, node):
        node.prev.next = node.next
        node.next.prev = node.prev
        self.size -= 1

class LFUCache:
    def __init__(self, capacity: int):
        self.capacity = capacity
        self.size = 0
        self.min_freq = 0
        self.cache = {}
        self.freq_map = collections.defaultdict(DLinkedList)

    def get(self, key: int) -> int:
        if key not in self.cache:
            return -1
        node = self.cache[key]
        self.update_freq(node)
        return node.value

    def put(self, key: int, value: int) -> None:
        if self.capacity == 0:
            return
        
        if key in self.cache:
            node = self.cache[key]
            node.value = value
            self.update_freq(node)
        else:
            if self.size >= self.capacity:
                old_list = self.freq_map[self.min_freq]
                del self.cache[old_list.tail.prev.key]
                old_list.remove_node(old_list.tail.prev)
                self.size -= 1
            
            node = Node(key, value)
            self.cache[key] = node
            self.freq_map[1].add_node(node)
            self.min_freq = 1
            self.size += 1
    
    def update_freq(self, node):
        freq = node.freq
        self.freq_map[freq].remove_node(node)
        if freq == self.min_freq and self.freq_map[freq].size == 0:
            self.min_freq += 1
        
        node.freq += 1
        self.freq_map[node.freq].add_node(node)

Time Complexity: O(1)

All operations (get and put) are performed in constant time.

Space Complexity: O(capacity)

Space used is proportional to the capacity of the cache.

class Node {
    int key;
    int value;
    int freq;
    Node prev;
    Node next;
    
    public Node(int key, int value) {
        this.key = key;
        this.value = value;
        this.freq = 1;
    }
    
    public Node() {
        this.freq = 0;
    }
}

class DLinkedList {
    Node head;
    Node tail;
    int size;
    
    public DLinkedList() {
        head = new Node();
        tail = new Node();
        head.next = tail;
        tail.prev = head;
        size = 0;
    }
    
    public void addNode(Node node) {
        node.prev = head;
        node.next = head.next;
        head.next.prev = node;
        head.next = node;
        size++;
    }
    
    public void removeNode(Node node) {
        node.prev.next = node.next;
        node.next.prev = node.prev;
        size--;
    }
}

class LFUCache {
    private int capacity;
    private int size;
    private int minFreq;
    private Map cache;
    private Map freqMap;
    
    public LFUCache(int capacity) {
        this.capacity = capacity;
        this.size = 0;
        this.minFreq = 0;
        this.cache = new HashMap<>();
        this.freqMap = new HashMap<>();
    }
    
    public int get(int key) {
        if (!cache.containsKey(key)) {
            return -1;
        }
        Node node = cache.get(key);
        updateFreq(node);
        return node.value;
    }
    
    public void put(int key, int value) {
        if (capacity == 0) {
            return;
        }
        
        if (cache.containsKey(key)) {
            Node node = cache.get(key);
            node.value = value;
            updateFreq(node);
        } else {
            if (size >= capacity) {
                DLinkedList oldList = freqMap.get(minFreq);
                cache.remove(oldList.tail.prev.key);
                oldList.removeNode(oldList.tail.prev);
                size--;
            }
            
            Node node = new Node(key, value);
            cache.put(key, node);
            if (!freqMap.containsKey(1)) {
                freqMap.put(1, new DLinkedList());
            }
            freqMap.get(1).addNode(node);
            minFreq = 1;
            size++;
        }
    }
    
    private void updateFreq(Node node) {
        int freq = node.freq;
        DLinkedList oldList = freqMap.get(freq);
        oldList.removeNode(node);
        
        if (freq == minFreq && oldList.size == 0) {
            minFreq++;
        }
        
        node.freq++;
        if (!freqMap.containsKey(node.freq)) {
            freqMap.put(node.freq, new DLinkedList());
        }
        freqMap.get(node.freq).addNode(node);
    }
}

class LFUCache {
private:
    int capacity, size, minFreq;
    unordered_map cache;
    unordered_map freqMap;
    
    void updateFreq(Node* node) {
        int freq = node->freq;
        freqMap[freq]->removeNode(node);
        
        if (freq == minFreq && freqMap[freq]->size == 0) {
            minFreq++;
        }
        
        node->freq++;
        if (!freqMap[node->freq]) {
            freqMap[node->freq] = new DLinkedList();
        }
        freqMap[node->freq]->addNode(node);
    }
    
public:
    LFUCache(int capacity) {
        this->capacity = capacity;
        size = 0;
        minFreq = 0;
    }
    
    int get(int key) {
        if (!cache.count(key)) {
            return -1;
        }
        Node* node = cache[key];
        updateFreq(node);
        return node->value;
    }
    
    void put(int key, int value) {
        if (capacity == 0) {
            return;
        }
        
        if (cache.count(key)) {
            Node* node = cache[key];
            node->value = value;
            updateFreq(node);
        } else {

                oldList->removeNode(oldList->tail->prev);

                freqMap[1] = new DLinkedList();

        }
    }
};

        }
        const node = this.cache.get(key);

    put(key, value) {

            const node = this.cache.get(key);
            node.value = value;
            this.updateFreq(node);

public class DLinkedList {

    public Node Tail { get; set; }

        node.Prev.Next = node.Next;

        Size--;

Approach Explanation

The solution implements an LFU Cache using the following components and techniques:

  1. Key Components:
    • Node class for storing key-value pairs
    • Doubly linked list for O(1) operations
    • Frequency map for tracking usage
    • Cache map for quick access
  2. Data Structures:
    • HashMap for cache
    • HashMap for frequency map
    • Doubly linked lists
    • Custom Node class

Implementation Details:

  • Constant time operations
  • Frequency tracking
  • LRU within same frequency
  • Capacity management

Optimization Insights:

  • O(1) time complexity
  • Efficient node removal
  • Quick frequency updates
  • Minimal space overhead

Edge Cases:

  • Empty cache
  • Zero capacity
  • Frequency ties
  • Cache full
