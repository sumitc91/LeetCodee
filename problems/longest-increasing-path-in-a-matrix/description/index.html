329. Longest Increasing Path in a Matrix

Jump to Solution: Python Java C++ JavaScript C#

Problem Description

Given an m x n integers matrix, return the length of the longest increasing path in matrix.

From each cell, you can either move in four directions: left, right, up, or down. You may not move diagonally or move outside the boundary (i.e., wrap-around is not allowed).

Example 1:

Input: matrix = [[9,9,4],[6,6,8],[2,1,1]]
Output: 4
Explanation: The longest increasing path is [1, 2, 6, 9].

Example 2:

Input: matrix = [[3,4,5],[3,2,6],[2,2,1]]
Output: 4
Explanation: The longest increasing path is [3, 4, 5, 6]. Moving diagonally is not allowed.

Example 3:

Input: matrix = [[1]]
Output: 1

Constraints:

  • m == matrix.length
  • n == matrix[i].length
  • 1 <= m, n <= 200
  • 0 <= matrix[i][j] <= 2³¹ - 1

Solution

Python Solution

class Solution:
    def longestIncreasingPath(self, matrix: List[List[int]]) -> int:
        if not matrix:
            return 0
            
        m, n = len(matrix), len(matrix[0])
        dp = [[0] * n for _ in range(m)]
        
        def dfs(i: int, j: int) -> int:
            # If already computed
            if dp[i][j] != 0:
                return dp[i][j]
            
            # Initialize with 1 (path length of current cell)
            dp[i][j] = 1
            
            # Check all four directions
            directions = [(0, 1), (1, 0), (0, -1), (-1, 0)]
            for di, dj in directions:
                ni, nj = i + di, j + dj
                if (0 <= ni < m and 0 <= nj < n and 
                    matrix[ni][nj] > matrix[i][j]):
                    dp[i][j] = max(dp[i][j], dfs(ni, nj) + 1)
            
            return dp[i][j]
        
        # Try starting from each cell
        longest = 0
        for i in range(m):
            for j in range(n):
                longest = max(longest, dfs(i, j))
                
        return longest

Time Complexity: O(mn)

Each cell is visited exactly once due to memoization.

Java Solution

class Solution {
    private int m, n;
    private int[][] matrix;
    private int[][] dp;
    private static final int[][] DIRECTIONS = {{0, 1}, {1, 0}, {0, -1}, {-1, 0}};
    
    public int longestIncreasingPath(int[][] matrix) {
        if (matrix == null || matrix.length == 0) {
            return 0;
        }
        
        this.matrix = matrix;
        this.m = matrix.length;
        this.n = matrix[0].length;
        this.dp = new int[m][n];
        
        int longest = 0;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                longest = Math.max(longest, dfs(i, j));
            }
        }
        
        return longest;
    }
    
    private int dfs(int i, int j) {
        // If already computed
        if (dp[i][j] != 0) {
            return dp[i][j];
        }
        
        // Initialize with 1 (path length of current cell)
        dp[i][j] = 1;
        
        // Check all four directions
        for (int[] dir : DIRECTIONS) {
            int ni = i + dir[0];
            int nj = j + dir[1];
            if (isValid(ni, nj) && matrix[ni][nj] > matrix[i][j]) {
                dp[i][j] = Math.max(dp[i][j], dfs(ni, nj) + 1);
            }
        }
        
        return dp[i][j];
    }
    
    private boolean isValid(int i, int j) {
        return i >= 0 && i < m && j >= 0 && j < n;
    }
}

public class Solution {
    private int m, n;
    private int[,] dp;
    private static readonly (int, int)[] directions = {(0, 1), (1, 0), (0, -1), (-1, 0)};
    
    public int LongestIncreasingPath(int[][] matrix) {
        if (matrix == null || matrix.Length == 0) {
            return 0;
        }
        
        m = matrix.Length;
        n = matrix[0].Length;
        dp = new int[m, n];
        
        int longest = 0;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                longest = Math.Max(longest, DFS(matrix, i, j));
            }
        }
        
        return longest;
    }
    
    private int DFS(int[][] matrix, int i, int j) {
        // If already computed
        if (dp[i, j] != 0) {
            return dp[i, j];
        }
        
        // Initialize with 1 (path length of current cell)
        dp[i, j] = 1;
        
        // Check all four directions
        foreach (var (di, dj) in directions) {
            int ni = i + di;
            int nj = j + dj;
            if (IsValid(ni, nj) && matrix[ni][nj] > matrix[i][j]) {
                dp[i, j] = Math.Max(dp[i, j], DFS(matrix, ni, nj) + 1);
            }
        }
        
        return dp[i, j];
    }
    
    private bool IsValid(int i, int j) {
        return i >= 0 && i < m && j >= 0 && j < n;
    }
}
