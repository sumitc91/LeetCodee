149. Max Points on a Line

Hard

Problem Description

Given an array of points where points[i] = [xi, yi] represents a point on the X-Y plane, return the maximum number of points that lie on the same straight line.

Examples

Example 1:
Input: points = [[1,1],[2,2],[3,3]]
Output: 3

Example 2:
Input: points = [[1,1],[3,2],[5,3],[4,1],[2,3],[1,4]]
Output: 4

Python Solution


def maxPoints(points: List[List[int]]) -> int:
    n = len(points)
    if n <= 2:
        return n
    
    def slope(p1, p2):
        x1, y1 = p1
        x2, y2 = p2
        if x1 == x2:
            return float('inf')
        return (y2 - y1) / (x2 - x1)
    
    max_points = 0
    for i in range(n):
        slopes = {}
        for j in range(n):
            if i != j:
                s = slope(points[i], points[j])
                slopes[s] = slopes.get(s, 1) + 1
        if slopes:
            max_points = max(max_points, max(slopes.values()))
    
    return max_points

Java Solution


class Solution {
    public int maxPoints(int[][] points) {
        int n = points.length;
        if (n <= 2) {
            return n;
        }
        
        int maxPoints = 0;
        for (int i = 0; i < n; i++) {
            Map slopes = new HashMap<>();
            for (int j = 0; j < n; j++) {
                if (i != j) {
                    double slope = getSlope(points[i], points[j]);
                    slopes.put(slope, slopes.getOrDefault(slope, 1) + 1);
                }
            }
            if (!slopes.isEmpty()) {
                maxPoints = Math.max(maxPoints, Collections.max(slopes.values()));
            }
        }
        
        return maxPoints;
    }
    
    private double getSlope(int[] p1, int[] p2) {
        int x1 = p1[0], y1 = p1[1];
        int x2 = p2[0], y2 = p2[1];
        if (x1 == x2) {
            return Double.POSITIVE_INFINITY;
        }
        return (double)(y2 - y1) / (x2 - x1);
    }
}

C++ Solution


class Solution {
public:
    int maxPoints(vector>& points) {
        int n = points.size();
        if (n <= 2) {
            return n;
        }
        
        int maxPoints = 0;
        for (int i = 0; i < n; i++) {
            unordered_map slopes;
            for (int j = 0; j < n; j++) {
                if (i != j) {
                    double slope = getSlope(points[i], points[j]);
                    slopes[slope] = slopes.count(slope) ? slopes[slope] + 1 : 2;
                }
            }
            for (const auto& pair : slopes) {
                maxPoints = max(maxPoints, pair.second);
            }
        }
        
        return maxPoints;
    }
    
private:
    double getSlope(const vector& p1, const vector& p2) {
        int x1 = p1[0], y1 = p1[1];
        int x2 = p2[0], y2 = p2[1];
        if (x1 == x2) {
            return numeric_limits::infinity();
        }
        return static_cast(y2 - y1) / (x2 - x1);
    }
};

JavaScript Solution


/**
 * @param {number[][]} points

var maxPoints = function(points) {
    const n = points.length;
    if (n <= 2) {
        return n;
    }
    
    const getSlope = (p1, p2) => {
        const [x1, y1] = p1;
        const [x2, y2] = p2;
        if (x1 === x2) {
            return Infinity;
        }
        return (y2 - y1) / (x2 - x1);
    };
    
    let maxPoints = 0;
    for (let i = 0; i < n; i++) {
        const slopes = new Map();
        for (let j = 0; j < n; j++) {
            if (i !== j) {
                const slope = getSlope(points[i], points[j]);
                slopes.set(slope, (slopes.get(slope) || 1) + 1);
            }
        }
        if (slopes.size > 0) {
            maxPoints = Math.max(maxPoints, Math.max(...slopes.values()));
        }
    }
    
    return maxPoints;
};

C# Solution


public class Solution {
    public int MaxPoints(int[][] points) {
        int n = points.Length;
        if (n <= 2) {
            return n;
        }
        
        int maxPoints = 0;
        for (int i = 0; i < n; i++) {
            Dictionary slopes = new Dictionary();
            for (int j = 0; j < n; j++) {
                if (i != j) {
                    double slope = GetSlope(points[i], points[j]);
                    slopes[slope] = slopes.ContainsKey(slope) ? slopes[slope] + 1 : 2;
                }
            }
            if (slopes.Count > 0) {
                maxPoints = Math.Max(maxPoints, slopes.Values.Max());
            }
        }
        
        return maxPoints;
    }
    
    private double GetSlope(int[] p1, int[] p2) {
        int x1 = p1[0], y1 = p1[1];
        int x2 = p2[0], y2 = p2[1];
        if (x1 == x2) {
            return double.PositiveInfinity;
        }
        return (double)(y2 - y1) / (x2 - x1);
    }
}
